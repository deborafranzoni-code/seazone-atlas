## Restruturação : 

1/2- Criar uma estrutura que permite quando é realizada uma atualização na coluna project_version , aí será feita a cópia dos dados anteriores a atualizaçã da tabela `spot_building_units`  para a tabela  `spot_building_units_history`  

Colunas adicionadas na tabela `spot_building_units`   :  `project_version  , has_decor , decor_contract_value e is_active` 


```
----- spot_building_units_history \[ somente update - \]-----

CREATE TABLE szi.spot_building_units_backup AS

SELECT *, now() AS backup_at, current_user AS backup_by

FROM szi.spot_building_units WHERE false;

ALTER TABLE szi.spot_building_units
ADD COLUMN project_version TEXT;


CREATE TABLE IF NOT EXISTS szi.spot_building_units_history 
( id BIGSERIAL PRIMARY KEY, log JSONB NOT NULL );

 CREATE OR REPLACE FUNCTION szi.fn_log_spot_building_units_version() 
 RETURNS TRIGGER AS $$ BEGIN 
 -- Só executa se project_version mudar 
 IF NEW.project_version IS DISTINCT FROM OLD.project_version THEN
  INSERT INTO szi.spot_building_units_history (log)
    VALUES (
        jsonb_build_object(
            'operation', 'UPDATE_PROJECT_VERSION',
            'changed_at', now(),
            'changed_by', current_user,
            'old_data', to_jsonb(OLD),
            'new_project_version', NEW.project_version
        )
    );

END IF;

RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_log_project_version ON szi.spot_building_units;

CREATE TRIGGER trg_log_project_version
BEFORE UPDATE ON szi.spot_building_units
FOR EACH ROW
EXECUTE FUNCTION szi.fn_log_spot_building_units_version();

---index
CREATE INDEX idx_units_history_log_gin
ON szi.spot_building_units_history
USING GIN (log);

---Adcionar novas colunas 
ALTER TABLE szi.spot_building_units
ADD COLUMN has_decor BOOLEAN,
ADD COLUMN decor_contract_value NUMERIC(12,2),
ADD COLUMN is_active BOOLEAN;
```
